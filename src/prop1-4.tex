    $$
    We proof $\H$ by induction on $A$.
    
    \begin{itemize}

            $$
                \infer{
                    \ngr(A^\ngr\vee (\neg A)^\ngr) \vdash \R \vee \ngr \R

                \infer{
                    \R \vdash B^\ngr \wedge C^\ngr

    \end{itemize}

    \item We proof this property by induction on $A$ and we denote by $IH$ the induction hypothesis.
    \begin{itemize}
